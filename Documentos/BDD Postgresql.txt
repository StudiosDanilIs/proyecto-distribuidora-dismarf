-- 1. Tabla de Roles (Para separar Administradores de Supervisores de Almacén)
CREATE TABLE roles (
    id_rol SERIAL PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE,
    descripcion TEXT
);

-- 2. Tabla de Usuarios (El personal de Dismarf que usará la App Móvil)
CREATE TABLE usuarios (
    id_usuario SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    id_rol INT REFERENCES roles(id_rol),
    creado_en TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 3. Tabla de Cavas Cuarto (Los lugares físicos que vamos a monitorear con los ESP32)
CREATE TABLE cavas (
    id_cava SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    ubicacion VARCHAR(150),
    temp_minima_permitida DECIMAL(5,2) NOT NULL,
    temp_maxima_permitida DECIMAL(5,2) NOT NULL,
    estado VARCHAR(20) DEFAULT 'Activa'
);

-- 4. Tabla de Inventario (Los productos perecederos dentro de cada cava)
CREATE TABLE inventario (
    id_lote SERIAL PRIMARY KEY,
    producto VARCHAR(150) NOT NULL,
    cantidad INT NOT NULL,
    fecha_ingreso DATE NOT NULL,
    fecha_vencimiento DATE NOT NULL,
    id_cava INT REFERENCES cavas(id_cava),
    estado_calidad VARCHAR(50) DEFAULT 'Óptimo'
);
